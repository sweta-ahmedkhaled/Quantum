def grover_oracle_010(qc):
    # 1. Flip qubits 0 and 2 to |1> so the control works on '0' states
    qc.x(0)
    qc.x(2)
    
    # 2. Apply Multi-Controlled Z
    # This flips the phase if q0=1, q1=1, q2=1 (which corresponds to original 010)
    qc.h(2)
    qc.mcx([0, 1], 2) 
    qc.h(2)
    
    # 3. Uncompute the X gates
    qc.x(0)
    qc.x(2)

def diffuser(qc, n):
    # Apply H to all qubits
    qc.h(range(n))
    
    # Apply X to all qubits
    qc.x(range(n))
        
    # Apply Multi-Controlled Z
    qc.h(n-1)
    qc.mcx(list(range(n-1)), n-1)
    qc.h(n-1)
    
    # Apply X to all qubits
    qc.x(range(n))
        
    # Apply H to all qubits
    qc.h(range(n))

def run_grover(iterations):
    n = 3
    qc = QuantumCircuit(n, n)
    
    # 1. Initialization: Uniform Superposition
    qc.h(range(n))
        
    # 2. Grover Iterations
    for _ in range(iterations):
        grover_oracle_010(qc)
        diffuser(qc, n)
        
    # 3. Measurement
    qc.measure(range(n), range(n))
    
    # 4. Simulator
    simulator = AerSimulator()
    compiled_circuit = transpile(qc, simulator)

    # Run the job
    job = simulator.run(compiled_circuit, shots=10000)
    result = job.result()
    counts = result.get_counts(qc)
    
    # Calculate probability
    marked_count = counts.get('010', 0)
    prob = marked_count / 10000
    
    # Returning the probabilities, counts, and circuit
    return prob, counts, qc
